#/bin/bash

# Listar Processos Jboss
process1=$(ps -ef | grep "Process Controller")
process2=$(ps -ef | grep "Host Controller")
process3=$(ps -ef | grep "Server:server-one")
process4=$(ps -ef | grep "Server:server-two")

# Kill Processos Process Controller e Host Controller Jboss
result1=`echo $process1 | awk '{print $2} '`
result2=`echo $process2 | awk '{print $2} '`
result3=`echo $process3 | awk '{print $2} '`
result4=`echo $process4 | awk '{print $2} '`

sudo kill -9 $result1
sudo kill -9 $result2
sudo kill -9 $result3
sudo kill -9 $result4
